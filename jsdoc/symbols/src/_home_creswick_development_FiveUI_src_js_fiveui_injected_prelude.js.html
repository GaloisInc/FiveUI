 48 /**
 49  * fiveui.query is a wrapper around the jQuery $() function.
 50  *
 51  * @description
 52  * fiveui.query recurses into iframes and frames, whereas $(...) stops
 53  * when it encounters internal frames.
 54  *
 55  * Generally, rules written for FiveUI will want to cover the entire
 56  * visible page, and as such, should use fiveui.query; however, $(...)
 57  * is available if recursing into frames is not necessary.
 58  *
 59  * @param {String} sel The jQuery selector string.
 60  * @param {Object} [context] The context to run the query within. This is often a DOM object/tree.
 61  * @returns {Object} A jQuery object, suitable for chaining.
 62  */
 63 fiveui.query = function (sel, context) {
 64   var ctx = context || document;
 65   var $results = jQuery(sel, ctx);
 66 
 67   jQuery('iframe, frame', ctx).each(
 68     function(idx, elt) {
 69       var $tempResults;
 70       if (elt.contentDocument) {
 71         try {
 72           $tempResults = fiveui.query(sel, elt.contentDocument);
 73         } catch (e) {
 74           console.log("encoutered a non-cooperative iframe/frame at " + $(elt).attr("src"));
 75           console.log(e.toString());
 76           $tempResults = [];
 77         }
 78 
 79         $results = $results.add($tempResults);
 80       }
 81     }
 82   );
 83 
 84   $filteredResults = $results.not('#fiveui-top')
 85                              .not('#fiveui-top *')
 86                              .filter(':visible');
 87 
 88   // update global stats
 89   fiveui.stats.numElts += $filteredResults.length;
 90 
 91   return $filteredResults;
 92 };

127 /**
128  * Non-destructively removes whitespace from the start and end of a
129  * string.
130  *
131  * @param {String} [s] The string to trim of whitespace.
132  * @returns {String} The input string, without leading or trailing
133  * whitespace. Returns null if you gave it null.
134  */
135 fiveui.string.trim = function(s) {
136   if (s) {
137     return s.replace(/^\s+|\s+$/g,"");
138   }
139   return s;
140 };

142 /**
143  * Tokenize a string on whitespace.
144  *
145  * @param {!String} s The string to tokenize.
146  * @returns {String[]>} An array of substrings.
147  */
148 fiveui.string.tokens = function (s) {
149   var posLength = function(ar) {
150     return 1 <= ar.length;
151   };
152 
153   return s.split(/\s/).filter(posLength);
154 };

157 /**
158  * A simple heuristic check to see if a string is in Title Case.
159  *
160  * @description
161  * This does not perform an exhaustive gramatical analysis, and as
162  * such, it is prone to generating false-positives in some cases.  In
163  * particular, it only has a short 'white list' of articles,
164  * conjections, and prepositions that are allowed to be in lower case.
165  *
166  * @param {!String} str The string to check.
167  * @returns {!boolean} true if the string is in title case, false if
168  * it is not.
169  */
170 fiveui.string.isTitleCase = function(str) {
171   var exception = function(str) {
172     var exceptions = [ 'a', 'an', 'the' // articles
173                      , 'and', 'but', 'for', 'not', 'or' // conjuctions
174                      , 'in', 'on' // short prepositions
175                      , 'to' ];
176     return exceptions.indexOf(str.toLowerCase()) != -1;
177   };
178 
179   if ( !fiveui.word.capitalized(str[0]) ) {
180     return false;
181   }
182 
183   var tokens = fiveui.string.tokens(str);
184   for (var i=1; i < tokens.length; ++i) {
185     if (!exception(tokens[i]) && !fiveui.word.capitalized(tokens[i])) {
186       return false;
187     }
188   }
189   return true;
190 };

203 /**
204  * Check to see if a sting begins with a capital letter.
205  *
206  * @param {!String} word The string to check for capitalization.
207  * @returns {!boolean}
208  */
209 fiveui.word.capitalized = function(word) {
210   return fiveui.isString(word) && word.search(/^[A-Z]/, word) >= 0;
211 };
212 
213 /**
214  * Check to see if a sting consists entirely of capital letters.
215  *
216  * @param {!String} word The string to check for capitalization.
217  * @returns {!boolean}
218  */
219 fiveui.word.allCaps = function(word) {
220   return fiveui.isString(word)
221     && word.search(/^\w/, word) >= 0
222     && (word == word.toUpperCase());
223 };

233 /**
234  * Color check compiler. It is recommended to use the jQuery plugin
235  * fiveui.jquery.cssIsNot instead.
236  *
237  * @param {!String} selector The HTML element selector to check.
238  * @param {String[]}  colorSet An array of strings containing the HEX values of
239  *                           colors in the desired color set.
240  * @returns {!function()}      A function which checks the rule
241  * @see fiveui.jquery.cssIsNot
242  */
243 fiveui.color.colorCheck = function (selector, colorSet) {
244   var allowable, i, fnStr, forEachFuncStr;
245   allowable = {};
246   for (i = 0; i < colorSet.length; i += 1) { allowable[colorSet[i]] = true; }
247   forEachFuncStr  = 'function (j, elt) {\n'
248                   + '  var allowable = ' + JSON.stringify(allowable) + ';\n'
249                   + '  var color = fiveui.color.colorToHex($(elt).css("color"));\n'
250                   + '  if (!(color in allowable)) {\n'
251                   + '    report("Disallowed color " + color + " in element matching " + ' + JSON.stringify(selector) + ', $(elt));\n'
252                   + '  }\n'
253                   + '}\n';
254   fnStr = 'function () { fiveui.query("' + selector + '").each(' + forEachFuncStr + '); }';
255   return eval('false||'+fnStr); // the `false||` trick is required for eval to parse a
256                                 // function expression ?!?
257 };

259 componentToHex = function (c) {
260   var hex = c.toString(16).toUpperCase();
261   return hex.length == 1 ? "0" + hex : hex;
262 }
263 
264 shortHexToHex = function (color) {
265   var have = color.length - 1;
266   var haveDigits = color.substr(1, color.length);
267   var need = 6 - have;
268   var reps = Math.ceil(need / have);
269   var i, strColor;
270   for (i = 0, stdColor = color; i < reps; i += 1) { stdColor += haveDigits; }
271   return stdColor.substr(0, 7);
272 };

274 // Compare RGBA objects
275 equalRGBA = function (c1, c2) {
276   return (c1.r == c2.r &&
277           c1.g == c2.g &&
278           c1.b == c2.b &&
279           c1.a == c2.a);
280 };

282 /**
283  * Convert RGB values to Hex.
284  */
285 fiveui.color.rgbToHex = function (r, g, b) {
286   return "#" + componentToHex(r) + componentToHex(g) + componentToHex(b);
287 }

289 /**
290  * Convert a 3-byte hex value to base-10 RGB 
291  */
292 fiveui.color.hexToRGB = function (hex) {
293     var result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
294     return result ? {
295         r: parseInt(result[1], 16),
296         g: parseInt(result[2], 16),
297         b: parseInt(result[3], 16)
298     } : null;
299 }
300 
301 /**
302  * Covert rgb colors to hex and abreviated hex colors to their full 3 byte
303  * form.
304  *
305  * In case there are parse errors during the conversion, i.e. color values
306  * that are not understood, the input is returned unchanged.
307  *
308  * @param {!String} color The color string to convert. This should be either of the form rgb(...) or #...
309  * @returns {!String} The color string in #XXXXXX form
310  * @throws {ParseError} if the rgb color string cannot be parsed
311  */
312 fiveui.color.colorToHex = function(color) {
313     if (color.substr(0, 1) === '#') {
314       if (color.length === 7) {
315         return color;
316       }
317       else { // deal with #0 or #F7 cases
318         return shortHexToHex(color);
319       }
320     }
321     else {  // color == 'rgb...'
322       var c = fiveui.color.colorToRGB(color)
323       return fiveui.color.rgbToHex(c.r, c.g, c.b);
324     }
325 };

327 /**
328  * Covert color to RGB color object.
329  *
330  * @param {!String} color The color string to convert. This should be either of the form rgb(...) or #...
331  * @returns {!Object} An RGB color object with attributes: r, g, b, a
332  * @throws {ParseError} if the rgb color string cannot be parsed
333  */
334 fiveui.color.colorToRGB = function(color) {
335 
336     if (color.substr(0, 1) === '#') {
337       return fiveui.color.hexToRGB(fiveui.color.colorToHex(color));
338     }
339 
340     var digits = /rgba?\((\d+), (\d+), (\d+)(, ([-+]?[0-9]*\.?[0-9]+))?/.exec(color);
341     if (!digits) {
342       throw new ParseError('could not parse color string: ' + color);
343     }
344 
345   var alpha = 1;
346   if (digits[5] != undefined) {
347     alpha = parseFloat(digits[5]);
348   }
349 
350   return { r: parseInt(digits[1]),
351            g: parseInt(digits[2]),
352            b: parseInt(digits[3]),
353            a: alpha };
354 };

356 /**
357  * Computationally determine the actual displayed background color for
358  * an object.  This accounts for parent colors that may appear when 
359  * a bg color is unspecified, or fully transparent.
360  *
361  * It does not account for elements that are shifted out of their
362  * parent containers.
363  *
364  * @param {!Object} A jquery object.
365  * @returns {color} an RGB color object. (no alpha - this does not
366  * return transparent colors)
367  */
368 fiveui.color.findBGColor = function(obj) {
369   var fc = fiveui.color;
370   var real = fc.colorToRGB(obj.css('background-color'));
371   var none = fc.colorToRGB('rgba(0, 0, 0, 0)');
372 
373   if (real.a != 1) {
374 
375     // find parents with a non-default bg color:
376     var parents = obj.parents().filter(
377       function() {
378         var color = fc.colorToRGB($(this).css('background-color'));
379         return !equalRGBA(color, none);
380       }).map(
381         function(i) {
382           return fc.colorToRGB($(this).css('background-color'));
383         });
384 
385     // push a white element onto the end of parents
386     parents.push({ r: 255, g: 255, b: 255, a: 1});
387 
388     // takeWhile alpha != 1
389     var colors = [];
390     for (var i=0; i < parents.length; i++) {
391       colors.push(parents[i]);
392       if (parents[i].a == 1) {
393         break;
394       }
395     }
396 
397     // Compose the colors and return. Note that fc.alphaCombine is
398     // neither commutative, nor associative, so we need to be carefull
399     // of the order in which parent colors are combined.
400     var res = real;
401     for (var i=0; i < colors.length; i++) {
402       res = fc.alphaCombine(res, colors[i]);
403     }
404     return res;
405   }
406   else {
407     return real;
408   }
409 };
410 
411 /**
412  * Combines two colors, accounting for alpha values less than 1.
413  * 
414  * @param {color} top The color "on top"
415  * @param {color} bot The color "on bottom"
416  * @return {color} the composite RGBA color.
417  */
418 fiveui.color.alphaCombine = function(top, bot) {
419   var result = {  };
420   result.r = Math.floor(top.r * top.a + bot.r * bot.a * (1 - top.a));
421   result.g = Math.floor(top.g * top.a + bot.g * bot.a * (1 - top.a));
422   result.b = Math.floor(top.b * top.a + bot.b * bot.a * (1 - top.a));
423 
424   result.a = top.a + bot.a * (1 - top.a);
425 
426   return result;
427 };

436 /**
437  * Extracts the font-family, font-size (in px, as an int), and font-weight
438  * from a jQuery object.
439  *
440  * @param {!Object} jElt A jQuery object to extract font info from
441  * @returns {!Object} An object with properties: 'family', 'size', and 'weight'
442  * @throws {ParseError} if the font size cannot be parsed
443  */
444 fiveui.font.getFont = function (jElt) {
445   var res   = {};
446   var size  = jElt.css('font-size');
447   if(size.length > 0) {
448     var psize = /(\d+)/.exec(size);
449     if(!psize) {
450       throw {
451         name: 'ParseError',
452         message: 'Could not parse font size: ' + jElt.css('font-size')
453       };
454     }
455     else {
456       res.size = psize
457     }
458   } else {
459     res.size = '';
460   }
461   res.family =  jElt.css('font-family');
462   res.weight =  jElt.css('font-weight').toString();
463   // normalize reporting of the following two synonyms
464   if (res.weight === '400') { res.weight = 'normal'; }
465   if (res.weight === '700') { res.weight = 'bold'; }
466   return res;
467 };

469 /**
470  * Validate a font property object extracted using fiveui.font.getFont().
471  *
472  * @description The `allow` parameter should be an object whose top level properties are
473  * (partial) font family names (e.g. 'Verdana'). For each font family name
474  * there should be an object whose properties are font weights (e.g. 'bold'),
475  * and for each font weight there should be an array of allowable sizes
476  * (e.g. [10, 11, 12]).
477  *
478  * The `font` parameter should be an object containing 'family', 'weight', and
479  * 'size' properties. These are returned by @see fiveui.font.getFont().
480  *
481  * @example > allow = { 'Verdana': { 'bold': [10, 12], 'normal': [10, 12]}};
482  * > font = { family: 'Verdana Arial sans-serif', size: "10", weight: "normal" };
483  * > fiveui.font.validate(allow, font) -> true
484  *
485  * @param {!Object} allow Object containing allowable font sets (see description and examples)
486  * @param {!Object} font object to check
487  * @param font.family A partial font family name (e.g. 'Verdana')
488  * @param font.weight A font weight (e.g. 'bold')
489  * @param font.size   A font size string (e.g. "10")
490  * @returns {!boolean}
491  */
492 fiveui.font.validate = function (allow, font) {
493   var x;
494   for (x in allow) { // loop over allowed font family keywords
495     if (font.family.indexOf(x) !== -1) { break; }
496     else { return false; }
497   }
498   return (font.weight in allow[x] &&  allow[x][font.weight].indexOf(parseInt(font.size)) != -1);
499 };

506 /**
507  * <p>Report a problem to FiveUI.</p>
508  *
509  * @description
510  * <p>report is used to indicate that a guideline has been violated.
511  * Invocations should provide a short (1-line) string description of
512  * the problem, as well as a reference to the element of the DOM that
513  * violated the guideline.</p>
514  *
515  * <p>The second parameter is not strictly required, but we strongly
516  * suggest providing a node if at all possible, as that is used to
517  * visually highlight the problematic areas of the DOM at runtime.
518  * Debugging a guideline violation becomes much more difficult without
519  * the visual indicator.</p>
520  *
521  * @function
522  * @param {!String} desc The description of the problem to report.
523  * @param {?Node} node The node in the DOM that caused the problem.
524  * @name report
525  */
526 
527 /**
528  * JSON.stringify is included here for extension and ruleset debugging
529  * purposes. It is used by the jQuery pluggin jQuery.fn.log.
530  */
531 var JSON = JSON || {};
532 JSON.stringify = JSON.stringify || function (obj) {
533     var t = typeof (obj);
534     if (t != "object" || obj === null) {
535         // simple data type
536         if (t == "string") obj = '"'+obj+'"';
537         return String(obj);
538     }
539     else {
540         // recurse array or object
541         var n, v, json = [], arr = (obj && obj.constructor == Array);
542         for (n in obj) {
543             v = obj[n]; t = typeof(v);
544             if (t == "string") v = '"'+v+'"';
545             else if (t == "object" && v !== null) v = JSON.stringify(v);
546             json.push((arr ? "" : '"' + n + '":') + String(v));
547         }
548         return (arr ? "[" : "{") + String(json) + (arr ? "]" : "}");
549     }
550 };
