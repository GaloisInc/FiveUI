 57 /**
 58  * fiveui.query is a wrapper around the jQuery $() function.
 59  *
 60  * @description
 61  * fiveui.query recurses into iframes and frames, whereas $(...) stops
 62  * when it encounters internal frames.
 63  *
 64  * Generally, rules written for FiveUI will want to cover the entire
 65  * visible page, and as such, should use fiveui.query; however, $(...)
 66  * is available if recursing into frames is not necessary.
 67  *
 68  * @param {String} sel The jQuery selector string.
 69  * @param {Object} [context] The context to run the query within. This is often a DOM object/tree.
 70  * @returns {Object} A jQuery object, suitable for chaining.
 71  */
 72 fiveui.query = function (sel, context) {
 73   var ctx = context || document;
 74   var $results = jQuery(sel, ctx);
 75 
 76   jQuery('iframe, frame', ctx).each(
 77     function(idx, elt) {
 78       var $tempResults;
 79       if (elt.contentDocument) {
 80         try {
 81           $tempResults = fiveui.query(sel, elt.contentDocument);
 82         } catch (e) {
 83           console.log("encoutered a non-cooperative iframe/frame at " + $(elt).attr("src"));
 84           console.log(e.toString());
 85           $tempResults = [];
 86         }
 87 
 88         $results = $results.add($tempResults);
 89       }
 90     }
 91   );
 92 
 93   $filteredResults = $results.not('#uic-top *')
 94                              .not('#uic-top')
 95                              .filter(':visible');
 96 
 97   // update global stats
 98   fiveui.stats.numElts += $filteredResults.length;
 99 
100   return $filteredResults;
101 };

136 /**
137  * Non-destructively removes whitespace from the start and end of a
138  * string.
139  *
140  * @param {String} [s] The string to trim of whitespace.
141  * @returns {String} The input string, without leading or trailing
142  * whitespace. Returns null if you gave it null.
143  */
144 fiveui.string.trim = function(s) {
145   if (s) {
146     return s.replace(/^\s+|\s+$/g,"");
147   }
148   return s;
149 };

151 /**
152  * Tokenize a string on whitespace.
153  *
154  * @param {!String} s The string to tokenize.
155  * @returns {String[]>} An array of substrings.
156  */
157 fiveui.string.tokens = function (s) {
158   var posLength = function(ar) {
159     return 1 <= ar.length;
160   };
161 
162   return s.split(/\s/).filter(posLength);
163 };

166 /**
167  * A simple heuristic check to see if a string is in Title Case.
168  *
169  * @description
170  * This does not perform an exhaustive gramatical analysis, and as
171  * such, it is prone to generating false-positives in some cases.  In
172  * particular, it only has a short 'white list' of articles,
173  * conjections, and prepositions that are allowed to be in lower case.
174  *
175  * @param {!String} str The string to check.
176  * @returns {!boolean} true if the string is in title case, false if
177  * it is not.
178  */
179 fiveui.string.isTitleCase = function(str) {
180   var exception = function(str) {
181     var exceptions = [ 'a', 'an', 'the' // articles
182                      , 'and', 'but', 'for', 'not', 'or' // conjuctions
183                      , 'in', 'on' // short prepositions
184                      , 'to' ];
185     return exceptions.indexOf(str.toLowerCase()) != -1;
186   };
187 
188   if ( !fiveui.word.capitalized(str[0]) ) {
189     return false;
190   }
191 
192   var tokens = fiveui.string.tokens(str);
193   for (var i=1; i < tokens.length; ++i) {
194     if (!exception(tokens[i]) && !fiveui.word.capitalized(tokens[i])) {
195       return false;
196     }
197   }
198   return true;
199 };

212 /**
213  * Check to see if a sting begins with a capital letter.
214  *
215  * @param {!String} word The string to check for capitalization.
216  * @returns {!boolean}
217  */
218 fiveui.word.capitalized = function(word) {
219   return fiveui.isString(word) && word.search(/^[A-Z]/, word) >= 0;
220 };
221 
222 /**
223  * Check to see if a sting consists entirely of capital letters.
224  *
225  * @param {!String} word The string to check for capitalization.
226  * @returns {!boolean}
227  */
228 fiveui.word.allCaps = function(word) {
229   return fiveui.isString(word)
230     && word.search(/^\w/, word) >= 0
231     && (word == word.toUpperCase());
232 };

243  * Color check compiler. It is recommended to use the jQuery plugin
244  * fiveui.jquery.cssIsNot instead.
245  *
246  * @param {!String} selector The HTML element selector to check.
247  * @param {String[]}  colorSet An array of strings containing the HEX values of
248  *                           colors in the desired color set.
249  * @returns {!function()}      A function which checks the rule
250  * @see fiveui.jquery.cssIsNot
251  */
252 fiveui.color.colorCheck = function (selector, colorSet) {
253   var allowable, i, fnStr, forEachFuncStr;
254   allowable = {};
255   for (i = 0; i < colorSet.length; i += 1) { allowable[colorSet[i]] = true; }
256   forEachFuncStr  = 'function (j, elt) {\n'
257                   + '  var allowable = ' + JSON.stringify(allowable) + ';\n'
258                   + '  var color = fiveui.color.colorToHex($(elt).css("color"));\n'
259                   + '  if (!(color in allowable)) {\n'
260                   + '    report("Disallowed color " + color + " in element matching " + ' + goog.json.serialize(selector) + ', $(elt));\n'
261                   + '  }\n'
262                   + '}\n';
263   fnStr = 'function () { fiveui.query("' + selector + '").each(' + forEachFuncStr + '); }';
264   return eval('false||'+fnStr); // the `false||` trick is required for eval to parse a
265                                 // function expression ?!?

268 /**
269  * Covert rgb colors to hex and abreviated hex colors to their full 3 byte
270  * form.
271  *
272  * In case there are parse errors during the conversion, i.e. color values
273  * that are not understood, the input is returned unchanged.
274  *
275  * @param {!String} color The color string to convert. This should be either of the form rgb(...) or #...
276  * @returns {!String} The color string in #XXXXXX form
277  * @throws {ParseError} if the rgb color string cannot be parsed
278  */
279 fiveui.color.colorToHex = function(color) {
280     var have, need;
281     if (color.substr(0, 1) === '#') {
282       if (color.length === 7) {
283         return color;
284       }
285       else { // deal with #0 or #F7 cases
286         var have = color.length - 1;
287         var haveDigits = color.substr(1, color.length);
288         var need = 6 - have;
289         var reps = Math.ceil(need / have);
290         var i, strColor;
291         for (i = 0, stdColor = color; i < reps; i += 1) { stdColor += haveDigits; }
292         return stdColor.substr(0, 7);
293       }
294     }
295 
296     var digits = /rgba?\((\d+), (\d+), (\d+)/.exec(color);
297     if (!digits) {
298       return color; // in case there is a parse error, we just
299                     // return the input unchanged
300     }
301 
302     var red = parseInt(digits[1]);
303     var green = parseInt(digits[2]);
304     var blue = parseInt(digits[3]);
305 
306     var rgb = blue | (green << 8) | (red << 16);
307     if (rgb === 0) {
308       return '#000000'; // normalized form
309     }
310     else {
311       return '#' + rgb.toString(16).toUpperCase();
312     }
313 };

323 /**
324  * Extracts the font-family, font-size (in px, as an int), and font-weight
325  * from a jQuery object.
326  *
327  * @param {!Object} jElt A jQuery object to extract font info from
328  * @returns {!Object} An object with properties: 'family', 'size', and 'weight'
329  * @throws {ParseError} if the font size cannot be parsed
330  */
331 fiveui.font.getFont = function (jElt) {
332   var res = {};
333   var sizeTxt = /(\d+)/.exec(jElt.css('font-size'));
334   if (!sizeTxt) {
335     throw {
336       name: 'ParseError',
337       message: 'Could not parse font size: ' + jElt.css('font-size')
338     };
339   }
340   else {
341     res.size = sizeTxt[1];
342   }
343   res.family =  jElt.css('font-family');
344   res.weight =  jElt.css('font-weight');
345   // normalize reporting of the following two synonyms
346   if (res.weight === '400') { res.weight = 'normal'; }
347   if (res.weight === '700') { res.weight = 'bold'; }
348   return res;
349 };

351 /**
352  * Validate a font property object extracted using fiveui.font.getFont().
353  *
354  * @description The `allow` parameter should be an object whose top level properties are
355  * (partial) font family names (e.g. 'Verdana'). For each font family name
356  * there should be an object whose properties are font weights (e.g. 'bold'),
357  * and for each font weight there should be an array of allowable sizes
358  * (e.g. [10, 11, 12]).
359  *
360  * The `font` parameter should be an object containing 'family', 'weight', and
361  * 'size' properties. These are returned by @see fiveui.font.getFont().
362  *
363  * @example > allow = { 'Verdana': { 'bold': [10, 12], 'normal': [10, 12]}};
364  * > font = { family: 'Verdana Arial sans-serif', size: "10", weight: "normal" };
365  * > fiveui.font.validate(allow, font) -> true
366  *
367  * @param {!Object} allow Object containing allowable font sets (see description and examples)
368  * @param {!Object} font object to check
369  * @param font.family A partial font family name (e.g. 'Verdana')
370  * @param font.weight A font weight (e.g. 'bold')
371  * @param font.size   A font size string (e.g. "10")
372  * @returns {!boolean}
373  */
374 fiveui.font.validate = function (allow, font) {
375   var x;
376   for (x in allow) { // loop over allowed font family keywords
377     if (font.family.indexOf(x) !== -1) { break; }
378     else { return false; }
379   }
380   return (font.weight in allow[x] &&  allow[x][font.weight].indexOf(parseInt(font.size)) != -1);
381 };

388 /**
389  * <p>Report a problem to FiveUI.</p>
390  *
391  * @description
392  * <p>report is used to indicate that a guideline has been violated.
393  * Invocations should provide a short (1-line) string description of
394  * the problem, as well as a reference to the element of the DOM that
395  * violated the guideline.</p>
396  *
397  * <p>The second parameter is not strictly required, but we strongly
398  * suggest providing a node if at all possible, as that is used to
399  * visually highlight the problematic areas of the DOM at runtime.
400  * Debugging a guideline violation becomes much more difficult without
401  * the visual indicator.</p>
402  *
403  * @function
404  * @param {!String} desc The description of the problem to report.
405  * @param {?Node} node The node in the DOM that caused the problem.
406  * @name report
407  */
408 
409 /**
410  * JSON.stringify is included here for extension and ruleset debugging
411  * purposes. It is used by the jQuery pluggin jQuery.fn.log.
412  */
413 var JSON = JSON || {};
414 JSON.stringify = JSON.stringify || function (obj) {
415     var t = typeof (obj);
416     if (t != "object" || obj === null) {
417         // simple data type
418         if (t == "string") obj = '"'+obj+'"';
419         return String(obj);
420     }
421     else {
422         // recurse array or object
423         var n, v, json = [], arr = (obj && obj.constructor == Array);
424         for (n in obj) {
425             v = obj[n]; t = typeof(v);
426             if (t == "string") v = '"'+v+'"';
427             else if (t == "object" && v !== null) v = JSON.stringify(v);
428             json.push((arr ? "" : '"' + n + '":') + String(v));
429         }
430         return (arr ? "[" : "{") + String(json) + (arr ? "]" : "}");
431     }
432 };
