 35 /**
 36  * Wrapper for the :contains('text') selector
 37  *
 38  * @param {!String} text Text to select for
 39  * @returns {!Object} A modified jQuery object
 40  */
 41 fiveui.jquery.hasText = function (text) {
 42   return this.filter(":contains('" + text + "')")
 43 }

 45 /**
 46  * Color checker plugin: filters for elements whose CSS color property is
 47  * not in the given set.
 48  *
 49  * @description Note: This is a special case of fiveui.jquery.cssIsNot, i.e.
 50  * $(..).notColorSet(set) == $(..).cssIsNot("color", set, fiveui.color.colorToHex)
 51  * @see {fiveui.color.colorToHex}
 52  *
 53  * @param {String[]} cset A set of allowable color strings
 54  * @returns {!Object} A modified jQuery object
 55  */
 56 fiveui.jquery.notColorSet = function (cset) {
 57   var allowable = {};
 58   for (var i = 0; i < cset.length; i += 1) { allowable[cset[i]] = true; } // array -> object
 59   return this.filter(function (index) {
 60     var color = fiveui.color.colorToHex($(this).css("color")); // .css("color") returns rgb(...)
 61     return !(color in allowable);
 62   });
 63 }

 65 /**
 66  * General CSS propetry checker plugin
 67  *
 68  * @description This plugin filters for elements whose CSS property `prop` is not a member
 69  * of the given array `cset`. The values checked are transformed using the
 70  * optional given function `fn`. This may be used to normalize values that the
 71  * browser returns so they can be compared to values in `cset`.
 72  *
 73  * @param {String} prop  CSS property selector
 74  * @param {String|String[]} set allowable values (either a string or an array of strings)
 75  * @param {function(String):String} [fn] Function to apply to return values of $(this).css(prop), fn defaults to the identity function.
 76  * @returns {Object} jQuery object
 77  */
 78 fiveui.jquery.cssIsNot = function (prop, set, fn) {
 79   var allowable = {};
 80   fn = fn || function (x) { return x; }; // default is Id
 81   if (typeof set === "string") {
 82     allowable[fn(set)] = true;
 83   }
 84   else { // assume `set` is an array of strings
 85    for (var i = 0; i < set.length; i += 1) { allowable[fn(set[i])] = true; } // array -> object
 86   }
 87   return this.filter(function (index) {
 88     var cssProp = fn($(this).css(prop));
 89     return !(cssProp in allowable);
 90   });
 91 }

 93 fiveui.jquery.linksTo = function (href) {
 94   return this.filter(function (index) {
 95     var addr = $(this).attr("href");
 96     return (addr == href);
 97   });
 98 }

100 /**
101  * Visually highlight elements in the jQuery object.
102  *
103  * @description This plugin is useful mostly in the process of writing
104  * guidelines, for example the guideline developer can load a page,
105  * click the "Break" button on the FiveUI window, enter the browser's
106  * Javascript console, and run:
107  *
108  * @example > $5("p").hasText("foo").highlight();
109  *
110  * @param {String} [hint] Highlighted border color, defaults to "red"
111  * @returns {!Object} A modified jQuery object
112  */
113 fiveui.jquery.highlight = function (hint) {
114   hint = hint || "red"; // Default is "red"
115   return this.css("background-color", "rgba(255, 0, 0, 0.3)")
116              .css("border-style", "solid")
117              .css("border-color", hint);
118 }

120 /**
121  * Returns a list of css properties that element in the jQuery
122  * object have.
123  *
124  * @description This plugin is useful for analysis of a given page when
125  * writing guielines. For example if the guideline developer wants to
126  * know what font sizes are used on a loaded page, they can run from the
127  * Javascript console:
128  *
129  * @example > $5("*").propDist("font-size", true);
130  *
131  * @param {String} prop CSS property to be inspected
132  * @param {boolean} [log] Boolean which enables console logging of the result; default is `false`.
133  * @returns {Object} A frequence map { "property": frequency }
134  */
135 fiveui.jquery.propDist = function (prop, log) {
136   var res = {};
137   log = log || false;
138   this.each(function (i, elt) {
139     var p = $(elt).css(prop);
140     if (p in res) {
141       res[p] += 1;
142     }
143     else {
144       res[p] = 1;
145     }
146   });
147   if (log) {
148     console.log("Property distribution:");
149     for (var p in res) {
150       console.log("  " + p + ": " + res[p]);
151     }
152   }
153   return res;
154 }

156 /**
157  * Register the plugins. This adds methods to the jQuery.fn namespace.
158  */
159 fiveui.jquery.init = function () {
160   for (fn in fiveui.jquery) {
161     f = fiveui.jquery[fn];
162     if (jQuery.isFunction(f) && fn != "init") {
163       jQuery.fn[fn] = fiveui.jquery[fn];
164     }
165   }
166 }
167 fiveui.jquery.init();
