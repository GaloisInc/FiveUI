 59 /**
 60  * fiveui.query is a wrapper around the jQuery $() function.
 61  *
 62  * fiveui.query recurses into iframes and frames, whereas $(...) stops
 63  * when it encounters internal frames.
 64  *
 65  * Generally, rules written for FiveUI will want to cover the entire
 66  * visible page, and as such, should use fiveui.query; however, $(...)
 67  * is available if recursing into frames is not necessary.
 68  *
 69  * @param {string} sel The jQuery selector string.
 70  * @param {?Object} context Optional: The context to run the query within.  This is often a DOM object/tree.
 71  * @return {Object} A jQuery object, suitable for chaining.
 72  */
 73 fiveui.query = function (sel, context) {
 74   var ctx = context || document;
 75   var $results = jQuery(sel, ctx);
 76 
 77   jQuery('iframe, frame', ctx).each(
 78     function(idx, elt) {
 79       if (elt.contentDocument) {
 80         $results = $results.add(fiveui.query(sel, elt.contentDocument));
 81       }
 82     }
 83   );
 84 
 85   return $results.not('#uic-top').filter(':visible');
 86 };

 90 /**
 91  * Determine if a given value is a string or not.
 92  *
 93  * @param {?*} o A value of some type that may or may not be defined.
 94  * @returns {!boolean} true if the object is a defined, non-null string, false
 95  * otherwise.
 96  */
 97 fiveui.isString = function(o) {
 98   return typeof o == 'string';
 99 };

109 /**
110  * Non-destructively removes whitespace from the start and end of a
111  * string.
112  *
113  * @param {?string} s The string to trim of whitespace.
114  * @return {?string} The input string, without leading or trailing
115  * whitespace. Returns null if you gave it null.
116  */
117 fiveui.string.trim = function(s) {
118   if (s) {
119     return s.replace(/^\s+|\s+$/g,"");
120   }
121   return s;
122 };
123 
124 /**
125  * Tokenize a string on whitespace.
126  *
127  * @param {!string} s The string to tokenize.
128  * @return {!Array.<!string>} An array of substrings.
129  */
130 fiveui.string.tokens = function (s) {
131   var posLength = function(ar) {
132     return 1 <= ar.length;
133   };
134 
135   return s.split(/\s/).filter(posLength);
136 };

139 /**
140  * A simple heuristic check to see if a string is in Title Case.
141  *
142  * This does not perform an exhaustive gramatical analysis, and as
143  * such, it is prone to generating false-positives in some cases.  In
144  * particular, it only has a short 'white list' of articles,
145  * conjections, and prepositions that are allowed to be in lower case.
146  *
147  * @param {!string} str The string to check.
148  * @returns {!boolean} true if the string is in title case, false if
149  * it is not.
150  */
151 fiveui.string.isTitleCase = function(str) {
152   var exception = function(str) {
153     var exceptions = [ 'a', 'an', 'the' // articles
154                      , 'and', 'but', 'for', 'not', 'or' // conjuctions
155                      , 'in', 'on' // short prepositions
156                      , 'to' ];
157     return exceptions.indexOf(str.toLowerCase()) != -1;
158   };
159 
160   if ( !fiveui.word.capitalized(str[0]) ) {
161     return false;
162   }
163 
164   var tokens = fiveui.string.tokens(str);
165   for (var i=1; i < tokens.length; ++i) {
166     if (!exception(tokens[i]) && !fiveui.word.capitalized(tokens[i])) {
167       return false;
168     }
169   }
170   return true;
171 };

173 /**
174  * Utilities for word-specific processing.
175  *
176  * The fiveui.word namespace focuses on tools for working directly
177  * with words in the sense of natural languages, rather than general
178  * strings (as is the case for the fiveui.string namespace).
179  *
180  * @namespace
181  */
182 fiveui.word = {};
183 
184 /**
185  * Check to see if a sting begins with a capital letter.
186  *
187  * @param {!string} word The string to check for capitalization.
188  * @returns {!boolean}
189  */
190 fiveui.word.capitalized = function(word) {
191   return fiveui.isString(word) && word.search(/^[A-Z]/, word) >= 0;
192 };
193 
194 /**
195  * Check to see if a sting consists entirely of capital letters.
196  * @param {!string} word The string to check for capitalization.
197  * @returns {!boolean}
198  */
199 fiveui.word.allCaps = function(word) {
200   return fiveui.isString(word)
201     && word.search(/^\w/, word) >= 0
202     && (word == word.toUpperCase());
203 };
